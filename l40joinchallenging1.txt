=>Active and 
SELECT CONCAT(firstname," ",lastname) AS 'active_members' FROM members
WHERE status = 1;

SELECT CONCAT(firstname," ",lastname) AS 'blocked_members' FROM members
WHERE status = 2;

SELECT CONCAT(firstname," ",lastname) AS name,
	CASE
		WHEN status = 1 THEN 'Active'
		ELSE "Blocked"
	END AS "Status"
FROM members;
